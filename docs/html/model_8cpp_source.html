170  fr_dx_susp = (volatile uint8_t) (fr_dx_susp + (SUSP_STROKE_NORMALIZE * filter_buffer(buf[obufn] + FR_DX_ADC_OFFSET, ADC_BUFFER_SIZE, ADC_CHANNELS))) / 2;
171 
172  if (calibrate) {
173  if (tps1_value < tps1_low) tps1_low = tps1_value;
174  if (tps1_value > tps1_max) tps1_max = tps1_value;
175  if (tps2_value < tps2_low) tps2_low = tps2_value;
176  if (tps2_value > tps2_max) tps2_max = tps2_value;
177  if (brake_value < brake_low) brake_low = brake_value;
178  if (brake_value > brake_max) brake_max = brake_value;
179  }
180 
181  tps1_percentage = map(tps1_value, tps1_low, tps1_max, 0, 100);
182  tps2_percentage = map(tps2_value, tps2_low, tps2_max, 0, 100);
183  brake_percentage = map(brake_value, brake_low, brake_max, 0, 100);
184 
185  // check APPS plausibility
186  if (abs(tps1_percentage - tps2_percentage) > APPS_PLAUS_RANGE)
187  apps_plausibility = false;
188  else
189  apps_plausibility = true;
190 
191  // check APPS + brake plausibility (BSPD)
192  if (tps1_percentage > 5 && brake_percentage > 25) // ACCELERATOR + BRAKE plausibility
193  brake_plausibility = false;
194  else if (!brake_percentage)
195  brake_plausibility = true;

203 void ADC_Handler() {
204  int f = ADC->ADC_ISR;
205  if (f & (1 << 27)){
206  bufn = (bufn + 1) &(BUFFERS - 1); // move DMA pointers to next buffer
207  ADC->ADC_RNPR = (uint32_t) buf[bufn];
208  ADC->ADC_RNCR = BUFFER_LENGTH;
